using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace estrutura_de_repeticao2
{
    class Program
    {
        static void Main(string[] args)
        {
            //uma variavel que tem varios valores do mesmo tipo = vetor; começa na posição 0; [para definir vetores]; exemplo: int[] idades= nem int [5];//
            // Exemplo Guarda 10 notas//
            int[] notas = new int[10];
            for (int i = 0; i < 10; i++)
            {
                Console.WriteLine("Digite a nota " + (i + 1));
                notas[i] = int.Parse(Console.ReadLine());
            }
            //encontrar o maior e o menor//
            int menor = notas[0];
            int maior = notas[0];
            for (int i = 1; i < 10; i++)
            {
                if (maior < notas[i]) maior = notas[i];
                if (menor > notas[i]) menor = notas[i];
            }
            //ver quantas vezes aparece a maior e a menor nota
            int contaMaior = 0;
            int contaMenor = 0;
            for (int i = 1; i < 10; i++)
            {
                if (maior == notas[i]) contaMaior++;
                if (menor == notas[i]) contaMenor++;
            }
            //encontrar a media das notas
            float soma = 0f;
            for (int i = 0; i < 10; i++)
            {
                //soma += notas[i]; ou soma = soma + notas[i];
                soma += notas[i];
            }
            float media = soma / 10;
            // exibir os resultados 
            Console.Clear();//Limpa a tela
            Console.WriteLine( "RESULTADOS");
            Console.WriteLine("O menor Valor é: " + menor);
            Console.WriteLine("O menor Valor aparece: " + contaMenor + " vezes");
            Console.WriteLine("O maior Valor é: " + maior);
            Console.WriteLine("O maior Valor aparece: " + contaMaior + " vezes");
            Console.WriteLine("a media das notas é  " + media);
            Console.ReadKey();
        }
    }
}
